dynamic verifyInput()

Verifies that the input expression is supported, right now the algorithm doesn't work with negative parenthesis ('-()')

Implementation

verifyInput() {
  _expression = _expression.replaceAll(' ', '');
  for (int i = 0; i < _expression.length; i++) {
    if (_expression[i] == "-") {
      if (i + 1 < _expression.length && _expression[i + 1] == "(") {
        if (i - 1 > 0 && _isOperator(_expression[i - 1])) {
          throw ExpressionException(
              "The actual algorithm can't handle negative parenthesis ('-()'), please rewrite the expression");
        }
      }
    }
  }
}
